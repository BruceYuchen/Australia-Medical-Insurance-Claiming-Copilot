### 2. 改进匹配算法
```python
def calculate_optimized_score(self, suggestion, query, query_features):
    # 基础分数
    base_score = suggestion.get('score', 0.0)
    
    # T1组别特殊处理
    if suggestion.get('group') == 'T1':
        t1_bonus = self._calculate_t1_bonus(query, suggestion['description'])
        base_score += t1_bonus
    
    # S类型特殊处理
    if suggestion.get('item_type') == 'S':
        s_bonus = self._calculate_s_type_bonus(query, suggestion['description'])
        base_score += s_bonus
    
    # 中文查询特殊处理
    if query_features['is_chinese']:
        chinese_bonus = self._calculate_chinese_bonus(query, suggestion['description'])
        base_score += chinese_bonus
    
    return min(1.0, base_score)
```
